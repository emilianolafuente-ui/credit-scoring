Pasos para configurar:
1. Ir a sheets.new y crear un nuevo Google Sheet.
2. En la hoja, ir a Extensiones → Apps Script.
3. Borrar el contenido y pegar el script de abajo. Guardarlo.
4. Hacer clic en Implementar → Nueva implementación → Aplicación web.
5. En "Ejecutar como" elegir Yo. En "Quién tiene acceso" elegir Cualquier persona.
6. Copiar la URL de implementación y pegarla abajo.
7. Autorizar los permisos cuando lo solicite.

function getPin_() { var s = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('_Config'); return s ? String(s.getRange('B1').getValue()) : ''; } function setPin_(pin) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var s = ss.getSheetByName('_Config') || ss.insertSheet('_Config'); s.getRange('A1').setValue('admin_pin'); s.getRange('B1').setValue(String(pin)); } function execAction(data) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName('Evaluaciones') || ss.insertSheet('Evaluaciones'); var action = data.action; if (action === 'save') { var row = data.row || data; if (sheet.getLastRow() === 0) { sheet.appendRow(['ID','Fecha','Cliente','País','Moneda','Período','Monto','Score', 'Nivel','Límite Aprobado','LMT','%','Fecha Revisión','Próx. Revisión','Veto','Nivel Auth','Score A','Score B','Analista','Detalle']); } else { var hdrs = sheet.getRange(1,1,1,sheet.getLastColumn()).getValues()[0]; if (hdrs.indexOf('Detalle') === -1) sheet.getRange(1, hdrs.length+1).setValue('Detalle'); } sheet.appendRow([String(row.id),row.fecha,row.nombre,row.pais,row.moneda,row.p2, row.monto,row.score,row.nivel,row.limiteAprobado,row.lmt,row.pct, row.fecha,row.proxRevision,row.anyVeto?'SÍ':'No','Nivel '+(row.nivelAuth+1),row.sA,row.sB,row.analista||'',row.detalle||'']); return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } if (action === 'delete') { var stored = getPin_(); if (stored && String(data.pin) !== stored) { return ContentService.createTextOutput(JSON.stringify({ok:false,err:'pin'})).setMimeType(ContentService.MimeType.JSON); } var id = data.id; var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(id)) { sheet.deleteRow(i+1); break; } } return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } if (action === 'clear') { var stored = getPin_(); if (stored && String(data.pin) !== stored) { return ContentService.createTextOutput(JSON.stringify({ok:false,err:'pin'})).setMimeType(ContentService.MimeType.JSON); } if (sheet.getLastRow() > 1) sheet.deleteRows(2, sheet.getLastRow()-1); return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } if (action === 'setPin') { var stored = getPin_(); if (stored && String(data.oldPin) !== stored) { return ContentService.createTextOutput(JSON.stringify({ok:false,err:'pin'})).setMimeType(ContentService.MimeType.JSON); } setPin_(data.newPin); return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ok:false,err:'unknown'})).setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { var data = JSON.parse(e.postData.contents); return execAction(data); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok:false,err:err.message})).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { if (e.parameter && e.parameter.action) { var data = {action: e.parameter.action}; var keys = ['id','pin','oldPin','newPin']; keys.forEach(function(k){ if(e.parameter[k]) data[k]=e.parameter[k]; }); if (e.parameter.data) { try { var p=JSON.parse(e.parameter.data); for(var k in p) data[k]=p[k]; } catch(err) {} } return execAction(data); } var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Evaluaciones'); if (!sheet || sheet.getLastRow() <= 1) { return ContentService.createTextOutput(JSON.stringify([])).setMimeType(ContentService.MimeType.JSON); } var rows = sheet.getDataRange().getValues(); var headers = rows[0]; var result = rows.slice(1).map(function(r) { var obj = {}; headers.forEach(function(h,i){ obj[h]=r[i]; }); return obj; }); return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON); Cómo subir a GitHub Pages (gratis, 5 minutos)
Paso 1: Crear cuenta gratuita en github.com
Paso 2: Crear un repositorio nuevo → darle un nombre (ej: credit-scoring)
Paso 3: Subir los archivos scoring_principal.html y formulario_cliente.html
Paso 4: Ir a Settings → Pages → Source: seleccionar main → Save
Paso 5: En 1–2 minutos tenés tu URL pública: tuusuario.github.io/credit-scoring/scoring_principal.html
Paso 6: Compartir esa URL con todos los analistas del equipo

Desde esa URL, Google Sheets funciona perfectamente y el historial es compartido.
